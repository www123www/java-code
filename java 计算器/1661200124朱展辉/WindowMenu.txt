package zzh;
import javax.swing.*;
import java.awt.event.InputEvent;
import java.awt.event.KeyEvent;
public class WindowMenu extends JFrame {
	
	
    JMenuBar menubar;
    JMenu menu,subMenu;
    JMenuItem jianQie,cheXiao,daKai,xinJian;
    JTextField text;
    Box boxW;
    Box boxH1,boxH2;
    PanelNullLayout panelNull ; //空布局的面板
    
    
    public WindowMenu(){} 
    public WindowMenu(String s,int x,int y,int w,int h) {
       init(s);
       setLocation(x,y);
       setSize(w,h);
       setVisible(true);
       setDefaultCloseOperation(JFrame.DISPOSE_ON_CLOSE); 
    }
    void init(String s){
       setTitle(s);             //设置窗口的标题   
       
       
       menubar=new JMenuBar(); 
       menu=new JMenu("文件");  
       xinJian=new JMenuItem("新建(N)");
       daKai=new JMenuItem("打开(O)");
       xinJian.setAccelerator(KeyStroke.getKeyStroke(KeyEvent.VK_N,InputEvent.CTRL_MASK)); 
       daKai.setAccelerator(KeyStroke.getKeyStroke(KeyEvent.VK_O,InputEvent.CTRL_MASK)); 
       menu.add(xinJian); 
       menu.add(daKai); 
       menubar.add(menu); 
       
       
       subMenu=new JMenu("编辑");  
       cheXiao=new JMenuItem("撤销(U)");
       jianQie=new JMenuItem("剪切(T)");
       cheXiao.setAccelerator(KeyStroke.getKeyStroke(KeyEvent.VK_Z,InputEvent.CTRL_MASK)); 
       jianQie.setAccelerator(KeyStroke.getKeyStroke(KeyEvent.VK_X,InputEvent.CTRL_MASK)); 
       subMenu.add(cheXiao); 
       subMenu.addSeparator();   //在菜单之间增加分隔线
       subMenu.add(jianQie); 
       menubar.add(subMenu); 
       setJMenuBar(menubar);
       
       
       panelNull=new PanelNullLayout();
       add(panelNull);
       
       
       
      
 
       
       
    } 
}
