package Test2;

import java.util.Scanner;

public class txlcs {
	public static void main(String[] args) {
		student[] stu = new student[3];
		Scanner sc = new Scanner(System.in);
		for (int i = 0; i < stu.length; i++) {
			student s = new student();
			System.out.println("请输入要录入的姓名：");
			s.setName(sc.nextLine());
			System.out.println("请输入要录入的电话号码");
			s.setPhone(sc.nextLine());
			stu[i] = s;
		}
		for (int i = 0; i < stu.length; i++) {
			System.out.println(stu[i].name + " " + stu[i].phone + "\n");
		}
		System.out.println("请输入要查询的姓名");
		String cx = sc.nextLine();
		Search(stu, cx);
	}

	private static void Search(student[] stu, String ss) {
		for (int i = 0; i < stu.length; i++) {
			if (ss.equals(stu[i].name)) {
				System.out.println(stu[i].name + "电话号码为" + stu[i].phone);
				break;
			} else if (i == stu.length - 1) {
				System.out.println("未找到" + ss + ",此人不存在于通讯录中");
			}
		}

	}
