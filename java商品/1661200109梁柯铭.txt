import java.util.Scanner;

public class Macket {
	Goods goods[];
	Macket(Goods goods[]){
		this.goods=goods;
	}
	public void chack(){
		for(int i=0;i<=goods.length;i++){
		System.out.println("商品名："+goods[i].getName()+"   价格："+goods[i].getPrice());}
			}
}

interface Goods {
	public abstract String getName();
	public abstract int getPrice();

}
class Compter implements Goods{
	String name;
	int price;
	Compter(String name,int price){
		this.name=name;
		this.price=price;
	}
	public String getName(){
		return name;
	}
	public int getPrice(){
		return price;
	}
}
 
 class Tv implements Goods{
	String name;
	int price;
	Tv(String name,int price){
		this.name=name;
		this.price=price;
	}
	public String getName(){
		return name;
	}
	public int getPrice(){
		return price;
	}
}
 
 class Phone implements Goods{
	String name;
	int price;
	Phone(String name,int price){
		this.name=name;
		this.price=price;
	}
	public String getName(){
		return name;
	}
	public int getPrice(){
		return price;
	}
}
