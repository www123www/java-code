import java.util.Scanner;
public class Check {

	public static void main(String[] args) {

		System.out.println("输入需要检查的字符串");

		String regex="[a-zA-Z|0-9|_|$]+[a-zA-Z|_|$|]";

		Scanner re = new Scanner(System.in);

		String str = re.nextLine();
		if(str.matches(regex)){
			System.out.println(str+" 已通过");
		}
		else{
			System.out.println(str+" 存在不合法标识符");
}
	}

}
