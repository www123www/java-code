	{

		       }
			      break;

		        {
			       x--;

			       };break;

		       {

		       };
		
		break;
		case 3:if(y-1>=0)
		{
			y--;y++;System.out.println(name+"向南走了一步，当前位置为"+"("+x+","+y+")");
		}else
	       {
	    	   System.out.println(name+"停下休息了一会，当前位置为"+"("+x+","+y+")");
	       }break;
		}
	
	}
	
}
